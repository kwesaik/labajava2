import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Введите число N: ");
        int N = scanner.nextInt(); // Ввод значения для переменной N

        int sum = 0;
        for (int i = 1; i <= N; i++) {
            sum += i; // Изменяем значение переменной sum на каждой итерации цикла
        }
        
        System.out.println("Сумма всех чисел от 1 до " + N + " равна: " + sum); // Вывод результата
        scanner.close();
    }
}
