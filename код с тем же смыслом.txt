import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        // Создаем объект класса Scanner для ввода данных
        Scanner scanner = new Scanner(System.in);

        // Запрашиваем у пользователя ввод числа N
        System.out.print("Введите число N: ");
        int N = scanner.nextInt();

        // Переменная для хранения суммы
        int sum = 0;

        // Цикл для суммирования чисел от 1 до N включительно
        for (int i = 1; i <= N; i++) {
            // Добавляем текущее значение i к общей сумме
            sum += i;
        }

        // Выводим результат
        System.out.println("Сумма всех чисел от 1 до " + N + " равна: " + sum);

        // Закрываем сканнер
        scanner.close();
    }
}
